function generarPDF() { const { jsPDF } = window.jspdf; const doc = new jsPDF(); const presupuesto = JSON.parse(localStorage.getItem('presupuestoActual') || '[]'); const observaciones = document.getElementById('observacionesGenerales').value; const nombreEmpresa = localStorage.getItem('nombreEmpresa') || "Profesional"; const rifEmpresa = localStorage.getItem('rifEmpresa') || ""; const direccionEmpresa = localStorage.getItem('direccionEmpresa') || "San Cristóbal, Táchira"; const telefonoEmpresa = localStorage.getItem('telefonoWhatsApp') || ""; const tasaBCV = parseFloat(localStorage.getItem('tasaBCV')) || 517.96; const logoBase64 = localStorage.getItem('logoEmpresa'); if (presupuesto.length === 0) { alert("No hay partidas para generar el PDF"); return; } const cliente = prompt("Nombre del cliente:"); if (!cliente) return; const porcentajeImpuesto = parseFloat(localStorage.getItem('porcentajeImpuesto')) || 0; if (logoBase64) { try { doc.addImage(logoBase64, 'JPEG', 15, 15, 25, 25); } catch (e) { } } doc.setFontSize(16); doc.setFont("helvetica", "bold"); doc.text(nombreEmpresa.toUpperCase(), 45, 22); doc.setFont("helvetica", "normal"); doc.setFontSize(9); doc.text("Dirección: " + direccionEmpresa, 45, 28); doc.text("RIF: " + rifEmpresa, 45, 33); if (telefonoEmpresa) { doc.text("Teléfono: " + telefonoEmpresa, 45, 38); } doc.line(15, 48, 195, 48); const fechaActual = new Date(); const fechaFormateada = `${String(fechaActual.getDate()).padStart(2, '0')}/${String(fechaActual.getMonth() + 1).padStart(2, '0')}/${String(fechaActual.getFullYear()).slice(-2)}`; let y = 56; doc.setFontSize(12); doc.setFont("helvetica", "bold"); doc.text("Presupuesto para: " + cliente, 15, y); doc.setFont("helvetica", "normal"); doc.setFontSize(10); doc.text("Fecha: " + fechaFormateada, 15, y + 7); y += 20; doc.setFontSize(10); doc.setFont("helvetica", "bold"); doc.text("Descripción", 15, y); doc.text("Total", 170, y); doc.line(15, y + 2, 195, y + 2); y += 10; let subtotalUSD = 0; presupuesto.forEach((p) => { if (y > 250) { doc.addPage(); y = 20; } // 1. MÓDULO: en negrita, sin color doc.setFont("helvetica", "bold"); doc.setFontSize(9); doc.text(p.modulo.toUpperCase(), 15, y); y += 6; // 2. EQUIPO/SERVICIO: primero el nombre completo doc.setFont("helvetica", "normal"); doc.setFontSize(10); const trabajoFormateado = p.trabajo || ''; const trabajoLines = doc.splitTextToSize(trabajoFormateado, 140); doc.text(trabajoLines, 15, y); // Precio total a la derecha doc.setFont("helvetica", "bold"); doc.text("$" + p.totalUSD.toFixed(2), 170, y); y += (trabajoLines.length * 5) + 2; // 3. DETALLES: cantidad, unidad y precio unitario doc.setFont("helvetica", "normal"); doc.setFontSize(8); doc.text(`Cantidad: ${p.cantidad} ${unidadLinter(p.unidad)} | Precio Unitario: $${parseFloat(p.precioUnitario).toFixed(2)}`, 15, y); y += 10; subtotalUSD += p.totalUSD; }); if (y > 240) { doc.addPage(); y = 20; } y += 2; doc.line(15, y, 195, y); y += 10; doc.setFont("helvetica", "bold"); if (porcentajeImpuesto > 0) { const montoImpuesto = subtotalUSD * (porcentajeImpuesto / 100); const granTotalConImpuesto = subtotalUSD + montoImpuesto; doc.text(`SUBTOTAL: $${subtotalUSD.toFixed(2)} USD`, 120, y); y += 6; doc.text(`IMPUESTO (${porcentajeImpuesto}%): $${montoImpuesto.toFixed(2)} USD`, 120, y); y += 6; doc.text(`TOTAL GENERAL: $${granTotalConImpuesto.toFixed(2)} USD`, 120, y); y += 6; doc.text(`Equivalente BCV: Bs. ${(granTotalConImpuesto * tasaBCV).toLocaleString('es-VE', {minimumFractionDigits: 2, maximumFractionDigits: 2})}`, 120, y); } else { doc.text(`TOTAL GENERAL: $${subtotalUSD.toFixed(2)} USD`, 120, y); y += 6; doc.text(`Equivalente BCV: Bs. ${(subtotalUSD * tasaBCV).toLocaleString('es-VE', {minimumFractionDigits: 2, maximumFractionDigits: 2})}`, 120, y); } if (observaciones && observaciones.trim() !== "") { const splitNotas = doc.splitTextToSize(observaciones, 180); if (y + 20 + (splitNotas.length * 5) > 280) { doc.addPage(); y = 20; } else { y += 15; } doc.setFont("helvetica", "bold"); doc.text("Nota General:", 15, y); y += 6; doc.setFont("helvetica", "normal"); doc.setFontSize(9); doc.text(splitNotas, 15, y); } doc.save(`Presupuesto_${cliente.replace(/\s+/g, '_')}.pdf`); }
